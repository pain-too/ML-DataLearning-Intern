    def clean_data(self):
        """数据清洗：处理缺失值和异常值"""
        print("\n=== 数据清洗 ===")

    def calculate_descriptive_stats(self):
        """计算描述性统计"""
        print("\n=== 描述性统计 ===")

    def calculate_improvement(self):
        """计算注意力改善程度（后测与前测的差值）"""
        print("\n=== 注意力改善分析 ===")
        
        improvement_results = {}

            pre_col = f'{test_type}_mean_pre'
            
            for group_id in [1, 2, 3]:
                group_data = self.df[self.df['group'] == group_id]
                group_name = self.group_names[group_id]

                            # 舒尔特：差值 = 前测 - 后测，正值表示改善
                            diff = valid_data[pre_col] - valid_data[post_col]

                        mean_improvement = diff.mean()
                        improvements.append(mean_improvement)
                        valid_counts.append(len(valid_data))
                    else:
                        improvements.append(np.nan)
                        valid_counts.append(0)
                
                improvement_results[f"{test_type}_group{group_id}"] = {
                    'improvements': improvements,
                    'counts': valid_counts
                }
                
                print(f"\n{group_name} - {self.test_names[test_type]}改善情况:")

                    if not np.isnan(imp):
                        print(f"  第{i+1}次后测: 平均改善={imp:.2f}秒 (N={valid_counts[i]})")
        
        return improvement_results
    
    def perform_statistical_tests(self):
        """执行统计检验"""
        print("\n=== 统计检验结果 ===")
        
        test_results = {}
        
        # 1. 组内比较：每个组前后测差异
        print("\n1. 组内前后测比较（配对t检验）:")
        
        for test_type in ['schulte', 'stroop']:
            pre_col = f'{test_type}_mean_pre'
            
            for group_id in [1, 2, 3]:
                group_data = self.df[self.df['group'] == group_id]
                group_name = self.group_names[group_id]
                
                # 计算最后一次后测与前测的差异
                last_post_col = f'{test_type}_mean_8'
                
                valid_mask = group_data[[pre_col, last_post_col]].notna().all(axis=1)
                valid_data = group_data[valid_mask]
                
                if len(valid_data) > 1:
                    pre_scores = valid_data[pre_col]
                    post_scores = valid_data[last_post_col]
                    
                    # 执行配对t检验
                    t_stat, p_value = stats.ttest_rel(pre_scores, post_scores)
                    
                    print(f"\n{group_name} - {self.test_names[test_type]}:")
                    print(f"  前测均值: {pre_scores.mean():.2f}")
                    print(f"  后测均值: {post_scores.mean():.2f}")
                    print(f"  t值: {t_stat:.3f}, p值: {p_value:.3f}")
                    print(f"  显著性: {'显著' if p_value < 0.05 else '不显著'}")
                    
                    key = f"within_group_{test_type}_group{group_id}"
                    test_results[key] = {
                        't_stat': t_stat,
                        'p_value': p_value,
                        'n': len(valid_data),
                        'pre_mean': pre_scores.mean(),
                        'post_mean': post_scores.mean()
                    }
        
        # 2. 组间比较：最后一次后测的改善程度差异
        print("\n\n2. 组间改善程度比较（ANOVA）:")
        
        for test_type in ['schulte', 'stroop']:
            pre_col = f'{test_type}_mean_pre'
            last_post_col = f'{test_type}_mean_8'
            
            improvement_scores = []
            group_labels = []
            
            for group_id in [1, 2, 3]:
                group_data = self.df[self.df['group'] == group_id]
                
                valid_mask = group_data[[pre_col, last_post_col]].notna().all(axis=1)
                valid_data = group_data[valid_mask]
                
                if len(valid_data) > 0:
                    # 计算改善分数
                    improvements = valid_data[pre_col] - valid_data[last_post_col]
                    improvement_scores.extend(improvements.tolist())
                    group_labels.extend([group_id] * len(improvements))
            
            # 执行单因素ANOVA
            if len(improvement_scores) > 0:
                groups = []
                for group_id in [1, 2, 3]:
                    group_improvements = [imp for imp, label in zip(improvement_scores, group_labels) 
                                         if label == group_id]
                    if len(group_improvements) > 0:
                        groups.append(group_improvements)
                
                if len(groups) >= 2:
                    f_stat, p_value = stats.f_oneway(*groups)
                    
                    print(f"\n{self.test_names[test_type]}改善程度组间比较:")
                    print(f"  F值: {f_stat:.3f}, p值: {p_value:.3f}")
                    print(f"  组间差异: {'显著' if p_value < 0.05 else '不显著'}")
                    
                    # 如果ANOVA显著，进行事后检验（Tukey HSD）
                    if p_value < 0.05 and len(groups) == 3:
                        print("  进行事后检验（Tukey HSD）:")
                        from statsmodels.stats.multicomp import pairwise_tukeyhsd
                        
                        tukey_results = pairwise_tukeyhsd(improvement_scores, group_labels)
                        print(tukey_results)
                    
                    key = f"between_groups_{test_type}"
                    test_results[key] = {
                        'f_stat': f_stat,
                        'p_value': p_value,
                        'n_groups': len(groups)
                    }
        
        return test_results

                valid_mask = group_df[[pre_col, last_post_col]].notna().all(axis=1)
                valid_data = group_df[valid_mask]
                
                if len(valid_data) > 1:
                    pre_scores = valid_data[pre_col].values
                    post_scores = valid_data[last_post_col].values
                    
                    # 配对t检验
                    t_stat, p_value = stats.ttest_rel(pre_scores, post_scores)
                    
                    # 计算效应量（Cohen's d）
                    mean_diff = np.mean(pre_scores - post_scores)
                    std_diff = np.std(pre_scores - post_scores, ddof=1)
                    cohens_d = mean_diff / std_diff if std_diff != 0 else 0
                    
                    result = {
                        '测试类型': test_name,
                        '实验组': group_name,
                        '样本量': len(valid_data),
                        '前测均值': float(np.mean(pre_scores)),
                        '后测均值': float(np.mean(post_scores)),
                        '均值差': float(mean_diff),
                        't值': float(t_stat),
                        'p值': float(p_value),
                        'Cohen\'s d': float(cohens_d),
                        '显著性': p_value < 0.05
                    }
                    
                    self.stat_test_results['within_group'][f"{test_key}_group{group_id}"] = result
                    
                    # 打印结果
                    print(f"\n  {group_name} - {test_name}:")
                    print(f"    前测: {result['前测均值']:.2f} ± {np.std(pre_scores):.2f}")
                    print(f"    后测: {result['后测均值']:.2f} ± {np.std(post_scores):.2f}")
                    print(f"    差值: {result['均值差']:+.2f}秒")
                    print(f"    t({result['样本量']-1}) = {result['t值']:.3f}, "
                          f"p = {result['p值']:.3f}")
                    print(f"    Cohen's d = {result['Cohen\'s d']:.3f}")
                    print(f"    结果: {'显著' if result['显著性'] else '不显著'}")
        
        # 2. 组间比较（ANOVA）
        print("\n\n2. 组间改善程度比较（单因素ANOVA）:")
        
        for test_key, test_data in self.test_info.items():
            test_name = test_data['name']
            pre_col = f'{test_key}_mean_pre'
            last_post_col = f'{test_key}_mean_8'
            
            # 收集各组的改善数据
            all_improvements = []
            all_groups = []
            
            for group_id, group_data in self.group_info.items():
                group_df = self.df[self.df['group'] == group_id]
                
                valid_mask = group_df[[pre_col, last_post_col]].notna().all(axis=1)
                valid_data = group_df[valid_mask]
                
                if len(valid_data) > 0:
                    improvements = valid_data[pre_col] - valid_data[last_post_col]
                    all_improvements.extend(improvements.tolist())
                    all_groups.extend([group_id] * len(improvements))
            
            # 执行ANOVA
            if len(set(all_groups)) >= 2:
                # 准备各组数据
                group_data_dict = {}
                for group_id in self.group_info.keys():
                    group_vals = [imp for imp, gid in zip(all_improvements, all_groups) 
                                 if gid == group_id]
                    if len(group_vals) > 0:
                        group_data_dict[group_id] = group_vals
                
                # 执行单因素ANOVA
                if len(group_data_dict) >= 2:
                    f_stat, p_value = stats.f_oneway(*group_data_dict.values())
                    
                    result = {
                        '测试类型': test_name,
                        'F值': float(f_stat),
                        'p值': float(p_value),
                        '显著性': p_value < 0.05,
                        '各组样本量': {self.group_info[gid]['name']: len(vals) 
                                   for gid, vals in group_data_dict.items()}
                    }
                    
                    self.stat_test_results['between_groups'][test_key] = result
                    
                    # 打印结果
                    print(f"\n  {test_name}:")
                    print(f"    F({len(group_data_dict)-1}, "
                          f"{len(all_improvements)-len(group_data_dict)}) = {f_stat:.3f}")
                    print(f"    p = {p_value:.3f}")
                    print(f"    组间差异: {'显著' if result['显著性'] else '不显著'}")
                    
                    # 如果ANOVA显著，进行事后检验
                    if result['显著性'] and len(group_data_dict) >= 2:
                        print("    事后检验（Tukey HSD）:")
                        try:
                            tukey_result = pairwise_tukeyhsd(all_improvements, all_groups)
                            print(tukey_result)
                        except Exception as e:
                            print(f"    事后检验失败: {e}")

def extract_pretest_data(file_path):
    """
    从Excel文件中提取每个id的前测数据并返回JSON格式
    
    Parameters:
    -----------
    file_path : str
        Excel文件路径
        
    Returns:
    --------
    dict : 包含所有志愿者前测数据的字典
    """
    
    print(f"正在读取文件: {file_path}")
    
    # 检查文件是否存在
    if not os.path.exists(file_path):
        return {"error": f"文件不存在: {file_path}"}
    
    try:
        # 读取Excel文件
        df = pd.read_excel(file_path)
        print(f"成功读取数据，共 {len(df)} 条记录")
        
        # 提取前测数据
        pretest_data = []
        
        for _, row in df.iterrows():
            # 提取基本信息
            volunteer_id = str(row['id'])
            
            # 提取组别（从id的第一位）
            group_id = int(volunteer_id[0])
            group_name = {
                1: "丁香组",
                2: "薰衣草组",
                3: "雪松组"
            }.get(group_id, f"未知组{group_id}")
            
            # 提取前测数据
            schulte_pretest = row.get('schulte_mean_pre')
            stroop_pretest = row.get('stroop_mean_pre')
            
            # 处理缺失值
            if pd.isna(schulte_pretest):
                schulte_pretest = None
            if pd.isna(stroop_pretest):
                stroop_pretest = None
            
            # 构建志愿者数据
            volunteer_info = {
                "id": volunteer_id,
                "group_id": group_id,
                "group_name": group_name,
                "pretest_data": {
                    "schulte": {
                        "test_name": "舒尔特方格",
                        "mean_time": float(schulte_pretest) if schulte_pretest is not None else None,
                        "unit": "秒"
                    },
                    "stroop": {
                        "test_name": "Stroop实验",
                        "mean_time": float(stroop_pretest) if stroop_pretest is not None else None,
                        "unit": "秒"
                    }
                }
            }
            
            pretest_data.append(volunteer_info)
        
        # 按组别和id排序
        pretest_data.sort(key=lambda x: (x['group_id'], int(x['id'])))
        
        # 构建最终结果
        result = {
            "metadata": {
                "file_path": file_path,
                "total_volunteers": len(pretest_data),
                "data_extracted": "前测数据",
                "timestamp": pd.Timestamp.now().strftime("%Y-%m-%d %H:%M:%S")
            },
            "group_summary": {
                "丁香组": len([v for v in pretest_data if v['group_id'] == 1]),
                "薰衣草组": len([v for v in pretest_data if v['group_id'] == 2]),
                "雪松组": len([v for v in pretest_data if v['group_id'] == 3])
            },
            "volunteers": pretest_data
        }
        
        return result
        
    except Exception as e:
        return {"error": f"读取数据时出错: {str(e)}"}

    Parameters:

    data : dict

    """

        output_path = os.path.join(base_dir, "pretest_data.json")

        print(f"数据已保存到: {output_path}")
        return output_path
    except Exception as e:
        print(f"保存JSON文件时出错: {e}")
        return None

# 2. 提取组别
df['id'] = df['id'].fillna(0).astype(str)
df['group'] = df['id'].apply(lambda x: int(x[0]) if len(x) > 0 and x[0].isdigit() else 0)
df = df[df['group'].isin([1, 2, 3])]

group_names = {1: "丁香组", 2: "薰衣草组", 3: "雪松组"}

results = []

for test_key, test_cn_name in test_names.items():

                if significant:

                        print(f"结论: {group_cn_name}的{test_cn_name}表现有显著下降")
                else:
                    print(f"结论: {group_cn_name}的{test_cn_name}表现无显著变化")
                
                # 保存结果
                result = {
                    'group': group_cn_name,
                    'test': test_cn_name,
                    'n': len(pre_scores),
                    'pre_mean': pre_mean,
                    'pre_std': pre_std,
                    'post_mean': post_mean,
                    'post_std': post_std,
                    'improvement': mean_improvement,
                    'improvement_std': improvement_std,
                    't_stat': t_stat,
                    'p_value': p_value,
                    'significant': significant
                }
                results.append(result)

                # 子图1: 前后测对比条形图

                              color=['#3498db', '#2ecc71'], alpha=0.8)
                ax1.set_xticks(x_pos)
                ax1.set_xticklabels(['前测', '第一次后测'], fontsize=12)
                ax1.set_ylabel('平均用时（秒）', fontsize=12)
                ax1.set_title(f'{group_cn_name} - {test_cn_name}\n前后测对比', fontsize=14, fontweight='bold')
                ax1.grid(True, alpha=0.3, axis='y')
                
                # 在条形上添加数值
                for i, (mean, std) in enumerate(zip(means, stds)):
                    ax1.text(i, mean + std + 0.5, f'{mean:.1f}±{std:.1f}', 
                            ha='center', va='bottom', fontsize=10)
                
                # 子图2: 改善值分布
                ax2.boxplot(improvements, patch_artist=True, 
                           boxprops=dict(facecolor='#f39c12', alpha=0.7))
                ax2.scatter([1] * len(improvements), improvements,

                ax2.set_xticks([1])
                ax2.set_xticklabels(['改善值分布'], fontsize=12)
                ax2.set_ylabel('改善值（秒）', fontsize=12)
                ax2.set_title(f'改善分析\n平均改善: {mean_improvement:+.2f}秒', fontsize=14, fontweight='bold')

                stats_text += f"p值: {p_value:.3f}\n"

def extract_time_point(col_name):
    """从列名中提取时间点数字，pre返回0，其他返回数字"""
    if 'pre' in col_name:
        return 0
    else:
        # 提取_mean_后面的数字
        import re
        match = re.search(r'_mean_(\d+)', col_name)
        if match:
            return int(match.group(1))
        else:
            return None

def check_duplicates_with_format(df, test_cols, test_name):
    """
    检查同一受试者在不同时间点的数据是否重复，并按照指定格式报错
    
    参数:
    df: 数据框
    test_cols: 要检查的列列表
    test_name: 测试名称（用于输出信息）

                    value_to_timepoints[val].append(time_point)
        
        # 检查哪些值有重复的时间点

        print(f"详细报告已保存到: {report_path}")
        
        # 统计信息

        print(f"有行内重复数据的行数: {len(duplicate_rows)}")
        print(f"无重复数据的行数: {len(df) - len(duplicate_rows)}")
        print("=" * 60)
        
else:
    print(f"错误: 数据列数不足，只有 {len(column_names)} 列，但需要至少9列")

print("\n处理完成！")

    """

                print(f"   每增加一个时间单位，成绩平均改善 {time_effect:.4f} 单位")
        else:
            print(f"➖ 结论: {test_name}成绩随时间变化不显著 (β = {time_effect:.4f}, p = {time_p:.4f})")
            print(f"   未发现时间对成绩有显著影响")
        
        # 计算效应大小（部分R²）
        # 对于LMM，可以计算条件R²和边际R²
        try:
            # 计算边际R²（仅固定效应）
            y = group_data['score']
            y_pred_fixed = result.predict(group_data)
            ss_res_fixed = np.sum((y - y_pred_fixed) ** 2)
            ss_tot = np.sum((y - np.mean(y)) ** 2)
            marginal_r2 = 1 - (ss_res_fixed / ss_tot)
            
            print(f"\n📈 效应大小:")
            print(f"  边际R² (固定效应解释的方差): {marginal_r2:.4f}")
            print(f"  解释: 时间因素解释了成绩变异的{marginal_r2*100:.1f}%")
        except:
            pass
        
        return {
            'group': group_name,
            'test': test_name,
            'model': result,
            'time_effect': time_effect,
            'time_p': time_p,
            'intercept': fixed_effects['Intercept'],
            'n_subjects': group_data['id'].nunique(),
            'n_observations': len(group_data),
            'aic': result.aic,
            'bic': result.bic
        }
        
    except Exception as e:
        print(f"❌ LMM分析失败: {str(e)}")
        return None

        name = row['name']
        
        # 根据姓名判断组别
        if '丁香' in str(name):
            group = '丁香'
        elif '薰衣草' in str(name):
            group = '薰衣草'
        elif '雪松' in str(name):
            group = '雪松'
        else:
            group = '未知'
            continue
        
        # 提取所有时间点的数据
        for time_point in range(0, 9):  # 0-8
            time_col = str(time_point)
            if time_col in row.index:
                value = row[time_col]
                if pd.notna(value):  # 只保留有效数据
                    long_data.append({
                        'id': subject_id,
                        'name': name,
                        'group': group,
                        'time': time_point,
                        'score': float(value),
                        'task': task_name
                    })
    
    return pd.DataFrame(long_data)

def run_lmm_analysis(data, group_name, task_name):
    """
    运行LMM分析
    """
    print(f"\n{'='*60}")
    print(f"分析组别: {group_name} - {task_name}")
    print(f"{'='*60}")
    
    # 筛选特定组的数据
    group_data = data[data['group'] == group_name].copy()
    
    if len(group_data) == 0:
        print(f"警告: {group_name}组没有数据！")
        return None
    
    print(f"样本量: {len(group_data)} 个观测值")
    print(f"被试数量: {group_data['id'].nunique()} 人")
    print(f"时间点范围: {group_data['time'].min()} 到 {group_data['time'].max()}")
    
    # 中心化时间变量（可选，有助于解释）
    group_data['time_centered'] = group_data['time'] - group_data['time'].mean()
    
    # 运行LMM模型
    try:
        # 使用statsmodels的MixedLM
        model = MixedLM.from_formula(
            'score ~ time',
            groups='id',
            re_formula='1 + time',
            data=group_data
        )
        
        result = model.fit(method='lbfgs')
        
        # 输出结果
        print("\nLMM模型结果:")
        print(result.summary())
        
        # 提取关键信息
        fixed_effects = result.fe_params
        p_values = result.pvalues
        
        print(f"\n固定效应:")
        print(f"截距 (Intercept): {fixed_effects['Intercept']:.4f}, p值: {p_values['Intercept']:.4f}")
        print(f"时间效应 (time): {fixed_effects['time']:.4f}, p值: {p_values['time']:.4f}")
        
        # 判断是否显著
        alpha = 0.05
        time_p = p_values['time']
        
        if time_p < alpha:
            if fixed_effects['time'] < 0:
                print(f"✓ 时间效应显著为负 (p={time_p:.4f})，成绩随时间显著下降")
                trend = "显著下降"
            else:
                print(f"✓ 时间效应显著为正 (p={time_p:.4f})，成绩随时间显著上升")
                trend = "显著上升"
        else:
            print(f"✗ 时间效应不显著 (p={time_p:.4f})，成绩随时间无显著变化")
            trend = "无显著变化"
        
        # 计算效应量（伪R²）
        # 对于混合模型，计算条件R²和边际R²
        y = group_data['score']
        y_pred = result.predict(group_data)
        residuals = y - y_pred
        
        # 边际R²（仅固定效应）
        ss_total = np.sum((y - y.mean())**2)
        ss_residual = np.sum(residuals**2)
        marginal_r2 = 1 - (ss_residual / ss_total)
        
        print(f"边际R² (仅固定效应): {marginal_r2:.4f}")
        
        return {
            'group': group_name,
            'task': task_name,
            'intercept': fixed_effects['Intercept'],
            'time_effect': fixed_effects['time'],
            'time_p': time_p,
            'trend': trend,
            'marginal_r2': marginal_r2,
            'n_obs': len(group_data),
            'n_subjects': group_data['id'].nunique(),
            'model_result': result
        }
        
    except Exception as e:
        print(f"模型拟合失败: {e}")
        return None

def run_lmm_for_group(data, group_name, task_name):
    """
    对特定组运行LMM分析
    """
    print(f"\n{'='*60}")
    print(f"📊 正在分析: {group_name}组 - {task_name}任务")
    print(f"{'='*60}")
    
    # 筛选数据
    group_data = data[(data['group'] == group_name) & (data['task'] == task_name)].copy()
    
    if len(group_data) == 0:
        print(f"❌ 错误: {group_name}组没有{task_name}任务数据")
        return None
    
    # 基本统计
    n_subjects = group_data['subject_id'].nunique()
    n_obs = len(group_data)
    avg_obs_per_subject = n_obs / n_subjects
    
    print(f"被试人数: {n_subjects}")
    print(f"观测总数: {n_obs}")
    print(f"平均每人观测次数: {avg_obs_per_subject:.2f}")
    print(f"时间点范围: {group_data['time'].min()} 到 {group_data['time'].max()}")
    
    # 中心化时间变量（有助于模型稳定性）
    group_data['time_centered'] = group_data['time'] - group_data['time'].mean()
    
    # 尝试使用statsmodels的MixedLM
    try:
        from statsmodels.regression.mixed_linear_model import MixedLM
        
        # 准备数据
        X = pd.DataFrame({
            'Intercept': 1,
            'time': group_data['time_centered']
        })
        y = group_data['score']
        groups = group_data['subject_id']
        
        # 构建模型
        model = MixedLM(y, X, groups=groups, exog_re=X[['Intercept', 'time']])
        
        # 拟合模型
        result = model.fit(method='lbfgs', maxiter=1000)
        
        # 提取结果
        fixed_effects = result.fe_params
        p_values = result.pvalues
        
        print(f"\n📈 LMM分析结果:")
        print(f"截距 (基线成绩): {fixed_effects['Intercept']:.4f}")
        print(f"时间效应 (斜率): {fixed_effects['time']:.4f}")
        print(f"时间效应p值: {p_values['time']:.4f}")
        
        # 判断显著性
        alpha = 0.05
        time_effect = fixed_effects['time']
        time_p = p_values['time']
        
        if time_p < alpha:
            if time_effect < 0:
                significance = "显著下降"
                trend_symbol = "↘"
            else:
                significance = "显著上升"
                trend_symbol = "↗"
        else:
            significance = "无显著变化"
            trend_symbol = "→"
        
        print(f"趋势判断: {trend_symbol} {significance}")
        
        # 计算模型拟合指标
        y_pred = result.predict()
        residuals = y - y_pred
        ss_total = np.sum((y - y.mean())**2)
        ss_residual = np.sum(residuals**2)
        marginal_r2 = 1 - (ss_residual / ss_total)
        
        print(f"边际R²: {marginal_r2:.4f}")
        print(f"AIC: {result.aic:.2f}")
        print(f"BIC: {result.bic:.2f}")
        
        # 返回结果
        return {
            'group': group_name,
            'task': task_name,
            'n_subjects': n_subjects,
            'n_obs': n_obs,
            'intercept': fixed_effects['Intercept'],
            'time_effect': time_effect,

            'significance': significance,
            'trend_symbol': trend_symbol,
            'marginal_r2': marginal_r2,

            'model': result,

    except Exception as e:
        print(f"❌ LMM分析失败: {e}")
        print("尝试使用简化模型...")
        
        # 备用方案：使用线性回归

def perform_group_comparisons(results_dict, all_data):
    """
    进行组间比较分析
    """
    try:
        comparison_results = []
        
        # 对每个任务进行组间比较
        for task in ['Stroop', 'Schulte']:
            print(f"\n📊 {task}任务组间比较:")
            
            # 提取该任务的所有数据
            task_data = all_data[all_data['task'] == task].copy()
            
            if len(task_data) == 0:
                continue
            
            # 计算各组的描述性统计
            groups = ['丁香', '薰衣草', '雪松']
            desc_stats = []
            
            for group in groups:
                group_data = task_data[task_data['group'] == group]
                if len(group_data) > 0:
                    desc_stats.append({
                        '任务': task,
                        '组别': group,
                        '样本量': len(group_data),
                        '被试数': group_data['subject_id'].nunique(),
                        '平均成绩': group_data['score'].mean(),
                        '成绩标准差': group_data['score'].std(),
                        '最小成绩': group_data['score'].min(),
                        '最大成绩': group_data['score'].max(),
                        '中位数': group_data['score'].median()
                    })
            
            # 进行ANOVA分析（简单组间比较）
            from scipy import stats
            
            group_data_list = []
            group_labels = []
            
            for group in groups:
                group_scores = task_data[task_data['group'] == group]['score'].values
                if len(group_scores) > 0:
                    group_data_list.append(group_scores)
                    group_labels.append(group)
            
            if len(group_data_list) >= 2:
                # 方差齐性检验
                levene_stat, levene_p = stats.levene(*group_data_list)
                
                # 单因素方差分析
                f_stat, p_value = stats.f_oneway(*group_data_list)
                
                print(f"  方差齐性检验: Levene统计量={levene_stat:.3f}, p={levene_p:.3f}")
                print(f"  单因素方差分析: F({len(group_data_list)-1}, {sum(len(g) for g in group_data_list)-len(group_data_list)})={f_stat:.3f}, p={p_value:.3f}")
                
                # 如果显著，进行事后检验
                if p_value < 0.05:
                    print("  ✓ 组间存在显著差异")
                    
                    # Tukey HSD事后检验
                    try:
                        from statsmodels.stats.multicomp import pairwise_tukeyhsd
                        
                        # 准备数据
                        all_scores = []
                        all_groups = []
                        
                        for i, scores in enumerate(group_data_list):
                            all_scores.extend(scores)
                            all_groups.extend([group_labels[i]] * len(scores))
                        
                        tukey_result = pairwise_tukeyhsd(all_scores, all_groups, alpha=0.05)
                        
                        print("  Tukey HSD事后检验结果:")
                        print(tukey_result)
                        
                        # 保存事后检验结果
                        for i in range(len(tukey_result.groupsunique)):
                            for j in range(i+1, len(tukey_result.groupsunique)):
                                comparison_results.append({
                                    '任务': task,
                                    '比较组': f"{tukey_result.groupsunique[i]} vs {tukey_result.groupsunique[j]}",
                                    '均值差': tukey_result.meandiffs[i*len(tukey_result.groupsunique) + j - (i+1)*i//2],
                                    'p值': tukey_result.pvalues[i*len(tukey_result.groupsunique) + j - (i+1)*i//2],
                                    '是否显著': tukey_result.reject[i*len(tukey_result.groupsunique) + j - (i+1)*i//2]
                                })
                    except Exception as e:
                        print(f"  ⚠ Tukey检验失败: {e}")
                else:
                    print("  ✗ 组间无显著差异")
            
            # 保存描述性统计
            if desc_stats:
                desc_df = pd.DataFrame(desc_stats)
                comparison_results.extend(desc_stats)
        
        # 保存比较结果
        if comparison_results:
            comp_df = pd.DataFrame(comparison_results)

    except Exception as e:

def create_group_comparison_visualization(comp_df):
    """
    创建组间比较可视化
    """
    try:
        fig, axes = plt.subplots(1, 2, figsize=(16, 6))
        
        # 1. 各组平均成绩对比
        ax1 = axes[0]
        
        # 提取描述性统计
        desc_data = comp_df[comp_df['样本量'].notna()].copy()
        
        if not desc_data.empty:
            # 按任务和组别分组
            for task in desc_data['任务'].unique():
                task_data = desc_data[desc_data['任务'] == task]
                
                x = range(len(task_data))
                means = task_data['平均成绩'].values
                stds = task_data['成绩标准差'].values
                
                bars = ax1.bar(x, means, yerr=stds, 
                              capsize=5, alpha=0.7,
                              label=f'{task}任务')
                
                # 在柱子上添加数值
                for bar, mean in zip(bars, means):
                    height = bar.get_height()
                    ax1.text(bar.get_x() + bar.get_width()/2., height + stds[0]/2,
                            f'{mean:.2f}', ha='center', va='bottom')
            
            ax1.set_xlabel('组别')
            ax1.set_ylabel('平均成绩')
            ax1.set_title('不同香气组平均成绩对比')
            ax1.set_xticks(range(len(desc_data)))
            ax1.set_xticklabels(desc_data['组别'] + '-' + desc_data['任务'], rotation=45)
            ax1.legend()
            ax1.grid(True, alpha=0.3)
        
        # 2. 样本量对比
        ax2 = axes[1]
        
        if not desc_data.empty:
            tasks = desc_data['任务'].unique()
            groups = desc_data['组别'].unique()
            
            x = np.arange(len(groups))
            width = 0.35
            
            for i, task in enumerate(tasks):
                task_specific = desc_data[desc_data['任务'] == task]
                samples = []
                for group in groups:
                    group_data = task_specific[task_specific['组别'] == group]
                    if not group_data.empty:
                        samples.append(group_data['样本量'].values[0])
                    else:
                        samples.append(0)
                
                ax2.bar(x + i*width - width/2, samples, width, 
                       label=f'{task}任务', alpha=0.7)
            
            ax2.set_xlabel('组别')
            ax2.set_ylabel('样本量（观测次数）')
            ax2.set_title('各组成绩观测次数对比')
            ax2.set_xticks(x)
            ax2.set_xticklabels(groups)
            ax2.legend()
            ax2.grid(True, alpha=0.3)
        
        plt.tight_layout()
        plt.savefig('组间比较图.png', dpi=300, bbox_inches='tight')
        plt.show()

    except Exception as e:
        print(f"⚠ 组间比较可视化失败: {e}")

# ==================== 数据质量检查函数 ====================
def check_data_quality(all_data):
    """
    检查数据质量
    """
    print("\n🔍 数据质量检查")
    print("-"*40)
    
    # 检查缺失值
    missing_values = all_data.isnull().sum().sum()
    print(f"缺失值总数: {missing_values}")
    
    # 检查异常值（使用IQR方法）
    for task in all_data['task'].unique():
        task_data = all_data[all_data['task'] == task]['score']
        Q1 = task_data.quantile(0.25)
        Q3 = task_data.quantile(0.75)
        IQR = Q3 - Q1
        lower_bound = Q1 - 1.5 * IQR
        upper_bound = Q3 + 1.5 * IQR
        
        outliers = task_data[(task_data < lower_bound) | (task_data > upper_bound)]
        print(f"{task}任务异常值数量: {len(outliers)} ({len(outliers)/len(task_data)*100:.1f}%)")
    
    # 检查数据分布
    print("\n数据分布统计:")
    print(all_data.groupby(['task', 'group'])['score'].describe())
    
    # 检查时间点分布
    print("\n各时间点数据量:")
    time_dist = all_data.groupby(['task', 'group', 'time']).size().unstack(fill_value=0)
    print(time_dist)

def sensitivity_analysis(results_dict, all_data):
    """
    进行敏感性分析
    """
    print("\n🔬 敏感性分析")
    print("-"*40)
    
    # 分析不同时间点的影响
    for task in ['Stroop', 'Schulte']:
        print(f"\n{task}任务敏感性分析:")
        
        # 移除前测数据后的分析
        task_data_no_baseline = all_data[(all_data['task'] == task) & (all_data['time'] > 0)].copy()
        
        if len(task_data_no_baseline) > 0:
            print("  移除前测数据后:")
            
            for group in ['丁香', '薰衣草', '雪松']:
                group_data = task_data_no_baseline[task_data_no_baseline['group'] == group]
                
                if len(group_data) > 10:  # 确保有足够数据
                    # 简单线性回归
                    from scipy import stats
                    slope, intercept, r_value, p_value, std_err = stats.linregress(
                        group_data['time'], group_data['score']
                    )
                    
                    print(f"    {group}组: 斜率={slope:.3f}, p={p_value:.3f}, R²={r_value**2:.3f}")
